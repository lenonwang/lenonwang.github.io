每天一道算法题-平移一个数组的右侧的几个数字到左侧

发表于 2023-05-26 更新于 2025-10-19

给定一个数组int[]a = {1,2,3,4,5,6,7},要求将数组变换为{5,6,7,1,2,3,4,5},通过一个算法来实现。要求时间复杂度为O(n),空间复杂度最小。

public class TransFormArray{
    public static void main(String[]args){
        int[] nums = {1,2,3,4,5,6,7};
        transformArray(nums,3);//向右移动
        transformArray(nums,-3);//向左移动
        for(int i=0;i<nums.length;i++){
            System.out.println(nums[i]+"、");
        }
    }

    /**
     * 当transLen>0右移，当transLen<0左移；
     **/
    private static int[] transFormArray(int[]nums,int transLen){
        int len = nums.length;
        for(int i=0;i< len;i++){
            //右移动三个位置,保持数组的长度是不变的，怎么实现呢？
            nums[(len+len+transLen+i)%len] = nums[i]
        }
    }
}

    public static void main(String[]args){

        reverse(nums);
        for(int = 0;i<nums.length;i++){
            System.out.println(nums[i]+"、");
        }
    }

    public static int[] reverse(int[] nums){
        int i=0;
        int j= nums.length-1;
        for(int i=0;i<nums.length;i++>){
            if(i<j){
            int temp = nums[i];
            nums[i++] = nums[j];
            nums[j--] = temp;
            }
        }
        return nums;
    }
}
